### 1.4 Create `src/i18n/index.ts`

```typescript
import i18n from "i18next";
import { initReactI18next } from "react-i18next";
import { invoke } from "@tauri-apps/api/core";
import en from "./en.json";
import zh from "./zh.json";

const SUPPORTED_LANGUAGES = ["en", "zh"];

// Detect the system language from the browser locale.
// Matches "zh", "zh-CN", "zh-Hans", "zh-Hans-CN" etc. → "zh".
// Everything else (including "zh-TW", "zh-Hant") → "en".
function detectSystemLanguage(): string {
  const locale = navigator.language || "";           // e.g. "zh-CN", "en-US"
  const primary = locale.split("-")[0].toLowerCase(); // "zh", "en", ...
  if (primary === "zh") {
    // Only treat Simplified Chinese locales as "zh".
    // Traditional Chinese (zh-TW, zh-Hant) falls back to English for now.
    const isTrad = /\b(TW|HK|Hant)\b/i.test(locale);
    return isTrad ? "en" : "zh";
  }
  return SUPPORTED_LANGUAGES.includes(primary) ? primary : "en";
}

// Read persisted language from the settings DB at startup.
// If no language has been saved yet, detect from the OS and persist it.
async function getInitialLanguage(): Promise<string> {
  try {
    const stored = await invoke<string | null>("get_setting", { key: "language" });
    if (stored && SUPPORTED_LANGUAGES.includes(stored)) return stored;

    // First launch — detect from system and persist so the backend
    // (AI system prompts) can also read it from the DB.
    const detected = detectSystemLanguage();
    await invoke("set_setting", { key: "language", value: detected }).catch(() => {});
    return detected;
  } catch {
    return detectSystemLanguage();
  }
}

// Exported promise so main.tsx can await it before rendering.
export const i18nReady = getInitialLanguage().then((lng) =>
  i18n.use(initReactI18next).init({
    resources: { en: { translation: en }, zh: { translation: zh } },
    lng,
    fallbackLng: "en",
    interpolation: { escapeValue: false },
  })
);

export default i18n;
```

**Key detail — system language detection:**
- On first launch (no `language` key in DB), we read `navigator.language`.
- If it matches Simplified Chinese (`zh`, `zh-CN`, `zh-Hans`, etc.), we default to `"zh"`. Traditional Chinese (`zh-TW`, `zh-HK`, `zh-Hant`) falls back to `"en"` since we don't have a Traditional Chinese translation yet.
- The detected language is immediately persisted to the DB via `set_setting` so the Rust backend (AI system prompts) can also read it without any additional plumbing.
- Once the user explicitly picks a language in Settings, that choice always wins.

## Risks & Mitigations

| Risk | Mitigation |
|---|---|
| Missing a hardcoded string | Search `src/` for quoted strings after migration; reviewer sweep |
| i18n init delays app startup | `get_setting` is a fast SQLite read; negligible delay |
| Chinese text breaks layout | Chinese is typically shorter than English; visual QA pass |
| AI ignores language instruction | Language instructions in system prompts are well-supported by all major providers |
| Module-level constants with strings can't use `t()` | Move them inside components or use getter functions |
